#include<iostream>
using namespace std;
int main ()
{
    int array[15], n, i, max, min;
    cout << "Unesite velicinu niza : ";
    cin >> n;
    cout << "Unesite sve elemente niza : ";
    for (i = 0; i < n; i++)
        cin >> array[i];
    max = array[0];
    for (i = 0; i < n; i++)
    {
        if (max < array[i])
            max = array[i];
    }
    min = array[0];
    for (i = 0; i < n; i++)
    {
        if (min > array[i])
            min = array[i];
    }
    cout << "Najveci broj u nizu je : " << max;
    cout << "\nNajmanji broj u nizu je : " << min;
    return 0;
}
